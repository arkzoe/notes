![[约束.png]]
```mysql
//添加外键
CREATE TABLE 表名(
	字段名 数据类型,
	[CONSTRAINT] [外键名称] FOREIGN KEY(外键字段名) REFERENCES 主表(主表列名)
);

ALTER TABLE 表名 ADD CONSTRAINT 外键名称 FOREIGN KEY(外键字段名) 主表(主表列名);
//删除外键
ALTER TABLE 表名 DROP FOREIGN  KEY 外键名称;
```
